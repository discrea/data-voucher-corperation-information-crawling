	function setCookie(name, value, expires) {
		var todayDate = new Date();
		todayDate.setDate(todayDate.getDate() + expires);
		document.cookie = name + '=' + escape(value) + '; path=/; expires=' + todayDate.toGMTString() + ';';
	}
	
	
	function checkPopCookie(cookieName, popId) {
		var cookies = document.cookie;
		
		if( cookies.indexOf(cookieName + '=done') < 0 ) {
			document.getElementById(popId).style.display = 'block';
		} else {
			document.getElementById(popId).style.display = 'none';
		}
	}

function checkActivity(timeout, interval, elapsed) {
	if ($.active) {
		elapsed = 0;
		$.active = false;
		$.get('heartbeat');
	}
	
	if (elapsed < timeout) {
		elapsed += interval;
		
		setTimeout(function() {
			checkActivity(timeout, interval, elapsed);
		}, interval);
	} else {
		$('.wrap_all').fadeOut();
		
		alert('장시간 미사용으로 자동 로그아웃 처리가 되었습니다.');
		
		
		$('#headerSiteType').val('datavoucher');
		$('#logoutForm').attr('action', '/datavoucher/mm/kdataLogout.do').submit();
	}
}

function fn_ctrgFilter(val) {
	var salCtgrFilter = $('#salCtgr').val();
	
	if (val == 'clear') {
		$('#salCtgr').val('');
		fn_setCtgrBtn('salClear');
	} else {
		var onClass = $('#saleTd' + val).attr('class');
		
		if (onClass == 'on') {
			var onIcon = $('#' + val).attr('class');
			var target = onIcon.replace('on', 'off');
			
			$('#saleTd' + val).attr('class', 'off');
			$('#' + val).attr('class', target);
			$('#' + val).removeAttr('title');
		}
		
		if (salCtgrFilter != '') {
			if ((salCtgrFilter.indexOf(val)) == 0) {
				salCtgrFilter = salCtgrFilter.replace(val, '').replace(',', '');
				$('#salCtgr').val(salCtgrFilter);
			} else if ((salCtgrFilter.indexOf(val)) == -1) {
				$('#salCtgr').val(salCtgrFilter + ',' + val);
			} else {
				salCtgrFilter = salCtgrFilter.replace(',' + val, '');
				$('#salCtgr').val(salCtgrFilter);
			}
		} else {
			$('#salCtgr').val(val);
		}
		
		fn_setCtgrBtn();
	}
}

function fn_saveIntrPrdc(rcpnYear, brno, sprnSctrCd, prdcId) {

		alert('로그인이 필요한 기능입니다.');
		$('#loginFrm').attr('action', '/datavoucher/cm/portalLoginPage.do').submit();
	} else {
		var params = {
			'rcpnYear': rcpnYear,
			'brno': brno,
			'sprnSctrCd': sprnSctrCd,
			'prdcId': prdcId,
		};
		
		$.ajax({
			method: 'POST',
			datatype: 'json',
			async: false,
			url: '/datavoucher/mp/ip/selectPintrPrdcVal.json',
			data: params,
			global: false,
			success: function(data) {
				var result = data.jsonView;

				}

		});
	}

function selectFile(smplId) {
	
	$('#fileDownload1').html('등록된 파일 정보가 없습니다.');
	
	var fileId = smplId;
	
	$.ajax({
		cache: false,
		url: '/datavoucher/file/fileList.do',
		dataType: 'json',
		data: {'fileId': fileId},
		success: function(data) {
			var str = '';
			var result = data.jsonView;
			
			for (var i = 0; i < result.length; i++) {
				str += '<div>';
				str += '<a href="#" onclick="fn_fileDownload(\'' + result[i]['fileId'] + '\',\'' + result[i]['seqno'] + '\'); return false;" title="다운로드"><img src="/datavoucher/images/dvpms/icon_notice_file.png" alt="다운로드" style="width: 15px;"> ' + result[i]['fileNm'] + '</a>';
				str += '</div>';
			}
			
			if (result.length == 0) {
				str += '등록된 파일 정보가 없습니다.';
			}
			
			$('#fileDownload1').empty();
			$('#fileDownload1').html(str);
		}
	});
}
